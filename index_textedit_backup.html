function normalizeSlot(v){

  return String(v||"").toUpperCase().replace(/\s+/g,"");

}


function normalizeTognr(v){

  return String(v||"").trim();

}


function isInvalidVehicleValue(v){

  const s = String(v || "").trim();

  return !s || s === "❌" || s === "-" || s.toLowerCase() === "ukjent";

}


function sanitizeVehicleValue(v){

  const s = String(v || "").trim();

  return isInvalidVehicleValue(s) ? "" : s;

}


function normalizeTimeString(v){

  const s = String(v || "").trim();

  const m = s.match(/\b(\d{1,2}):(\d{2})\b/);

  if(!m) return "";

  return `${m[1].padStart(2,"0")}:${m[2]}`;

}


function readTrainNo(obj){

  if(!obj || typeof obj !== "object") return "";

  return normalizeTognr(

    obj.train_no ??

    obj.trainNo ??

    obj.train ??

    obj.tognr ??

    obj.tognummer ??

    obj.number ??

    obj.no ??

    ""

  );

}


function readDepartureTime(obj){

  if(!obj || typeof obj !== "object") return "";

  const candidates = [

    obj.time,

    obj.departure,

    obj.departure_time,

    obj.departureTime,

    obj.advertised_time,

    obj.advertisedTime,

    obj.scheduled_time,

    obj.scheduledTime,

    obj.public_time,

    obj.publicTime

  ];


  for(const candidate of candidates){

    const t = normalizeTimeString(candidate);

    if(t) return t;

  }


  return "";

}

function buildDepartureMapFromPayload(payload){

  const map = {};

  const departures = Array.isArray(payload?.departures)

    ? payload.departures

    : Array.isArray(payload?.data?.departures)

      ? payload.data.departures

      : [];


  departures.forEach(dep=>{

    const tog = readTrainNo(dep);

    const time = readDepartureTime(dep);

    if(tog && time && !map[tog]){

      map[tog] = time;

    }

  });


  return map;

}


function buildVehicleMapFromPayload(payload){

  const vehiclesRaw = payload?.vehicles ?? payload?.data?.vehicles;

  const cleanedVehicles = {};


  if(Array.isArray(vehiclesRaw)){

    vehiclesRaw.forEach(item=>{

      if(!item || typeof item !== "object") return;

      const tog = readTrainNo(item);

      const value = sanitizeVehicleValue(

        item.vehicle ??

        item.vehicles ??

        item.material ??

        item.kjoretoy ??

        item.value ??

        ""

      );

      if(tog) cleanedVehicles[tog] = value;

    });

    return cleanedVehicles;

  }


  if(vehiclesRaw && typeof vehiclesRaw === "object"){

    Object.entries(vehiclesRaw).forEach(([key,val])=>{

      const tog = normalizeTognr(key);

      if(!tog) return;

      cleanedVehicles[tog] = sanitizeVehicleValue(val);

    });

    return cleanedVehicles;

  }


  throw new Error("JSON-fila mangler gyldig 'vehicles'-mapping.");

}


function importBaliseJsonObject(payload){

  if(!payload || typeof payload !== "object"){

    throw new Error("Ugyldig JSON-format");

  }


  const cleanedVehicles = buildVehicleMapFromPayload(payload);


  const mode = String(payload.mode || payload?.data?.mode || state.baliseSelectedMode || "").trim().toLowerCase();

  const date = String(payload.date || payload?.data?.date || "").trim();


  if(mode !== "idag" && mode !== "imorgen"){

    throw new Error("JSON-fila mangler gyldig mode ('idag' eller 'imorgen').");

  }


  const departureMap = buildDepartureMapFromPayload(payload);


  const meta = {

    source: payload.source || payload?.data?.source || "balise.no",

    mode,

    date,

    generated_at: payload.generated_at || payload?.data?.generated_at || "",

    hasDepartures: Object.keys(departureMap).length > 0

  };


  if(mode === "idag"){

    state.baliseToday = cleanedVehicles;

    state.baliseTodayDepartures = departureMap;

    state.baliseMetaToday = meta;

  }else{

    state.baliseTomorrow = cleanedVehicles;

    state.baliseTomorrowDepartures = departureMap;

    state.baliseMetaTomorrow = meta;

  }


  persist(true);

  buildOppstilling();

  updateBaliseStatus();

  updateBaliseViewState();

}


function parseWashCommand(v){

  const s=String(v||"").trim().toLowerCase();

  return {

    set1: s==="ja" || s==="ja1",

    set2: s==="ja" || s==="ja2"

  };

}


function mergeRowsIntoResult(rows,result,map){

  rows.forEach(row=>{

    if(row.type) return;


    const tog=String(row.b||"").trim();

    const wash=parseWashCommand(row.e);


    const mat1=String(row.c||"").trim();

    const spor1=normalizeSlot(row.f);

    if(mat1&&spor1&&map[spor1]!==undefined){

      result[map[spor1]].tog=tog;

      result[map[spor1]].mat=mat1;

      result[map[spor1]].wash=wash.set1;

    }


    const mat2=String(row.d||"").trim();

    const spor2=normalizeSlot(row.h);

    if(mat2&&spor2&&map[spor2]!==undefined){

      result[map[spor2]].tog=tog;

      result[map[spor2]].mat=mat2;

      result[map[spor2]].wash=wash.set2;

    }

  });

}


function computeInndata(){

  mergeRowsIntoResult(state.turneringKveld,result,map);

  mergeRowsIntoResult(state.turneringNatt,result,map);

  return result;

}

          sett2:"",

        };

function getAllRelevantTursattTrains(){

  const trains = new Set();


  function addTrain(tog){

    const t = normalizeTognr(tog);

    if(!t) return;

    if(t.toLowerCase() === "ikke tursatt") return;

    trains.add(t);

  }


  fixedOppstillingRows.forEach(r => addTrain(r.train));


  state.turneringKveld.forEach(r => {

    if(!r.type) addTrain(r.b);

  });


  state.turneringNatt.forEach(r => {

    if(!r.type) addTrain(r.b);

  });


  Object.keys(getApiDepartureMap()).forEach(addTrain);

  Object.entries(state.apiBalise?.arrivals || {}).forEach(([train, info]) => {

    if(info && info.nextDay) addTrain(train);

  });


  return trains;

}


function getRelevantTrainListForApi(){

    const na = Number(a);

    const nb = Number(b);


    if(!Number.isNaN(na) && !Number.isNaN(nb)) return na - nb;

    return String(a).localeCompare(String(b), "no");

  });

}


function sortTimeAndTrainEntries(a,b){

  const ta = a.sortMinutes ?? 9999;

  const tb = b.sortMinutes ?? 9999;


  if(ta !== tb) return ta - tb;


  const na = Number(a.train);

  const nb = Number(b.train);


  if(!Number.isNaN(na) && !Number.isNaN(nb)) return na - nb;


  return String(a.train).localeCompare(String(b.train), "no");

}


function buildDepartureRowsForOppstilling(){

  return Object.entries(getApiDepartureMap())

    .map(([train,time])=>{

      const cleanTime = normalizeTimeString(time);

      const [hh,mm] = cleanTime.split(":").map(Number);

      const sortMinutes = Number.isNaN(hh) || Number.isNaN(mm) ? 9999 : (hh * 60 + mm);


      return {

        train: normalizeTognr(train),

        time: cleanTime,

        displayTime: cleanTime,

        sortMinutes

      };

    })

    .filter(row=>row.train)

    .sort(sortTimeAndTrainEntries);

}


function buildArrivalRowsForOppstilling(){

  return Object.entries(getApiArrivalMap())

    .map(([train,info])=>{

      const cleanTime = normalizeTimeString(info?.time);

      const [hh,mm] = cleanTime.split(":").map(Number);

      const nextDay = Boolean(info?.nextDay);

      const baseMinutes = Number.isNaN(hh) || Number.isNaN(mm) ? 9999 : (hh * 60 + mm);

      const sortMinutes = baseMinutes === 9999 ? 9999 : baseMinutes + (nextDay ? 1440 : 0);


      return {

        train: normalizeTognr(train),

        time: cleanTime,

        displayTime: cleanTime ? `${cleanTime}${nextDay ? " +1" : ""}` : "",

        nextDay,

        sortMinutes

      };

    })

    .filter(row=>row.train)

    .sort(sortTimeAndTrainEntries);

}

function normalizeApiDepartureMap(raw){

  const map = {};

  if(!raw || typeof raw !== "object") return map;


  Object.entries(raw).forEach(([train, value])=>{

    const tog = normalizeTognr(train);

    const time = normalizeTimeString(typeof value === "string" ? value : value?.time);

    if(tog && time) map[tog] = time;

  });


  return map;

}


function normalizeApiArrivalMap(raw){

  const map = {};

  if(!raw || typeof raw !== "object") return map;


  Object.entries(raw).forEach(([train, value])=>{

    const tog = normalizeTognr(train);

    if(!tog || !value || typeof value !== "object") return;


    const time = normalizeTimeString(value.time);

    if(!time) return;


    map[tog] = {

      time,

      nextDay:Boolean(value.nextDay)

    };

  });


  return map;

}


function renderApiTables(){

  const departuresTable = document.getElementById("apiDeparturesTable");

  const arrivalsTable = document.getElementById("apiArrivalsTable");

  const apiStatus = document.getElementById("apiStatus");

  const apiMeta = document.getElementById("apiMeta");


  const api = state.apiBalise || {};

  const departures = api.departures || {};

  const arrivals = api.arrivals || {};


  if(!api.ok){

    departuresTable.innerHTML = "<tr><td>Ingen avganger fra API ennå</td></tr>";

    arrivalsTable.innerHTML = "<tr><td>Ingen ankomster fra API ennå</td></tr>";


    if(api.error){

      apiStatus.innerHTML = `<span class="api-error">${escapeHtml(api.error)}</span>`;

      apiMeta.textContent = "Klarte ikke å hente data fra /api/balise-vehicles.";

    }else{

      apiStatus.textContent = "API ikke lest ennå";

      apiMeta.textContent = "Ingen data fra /api/balise-vehicles ennå.";

    }

    return;

  }


  apiStatus.textContent = `API lest: ${Object.keys(departures).length} avganger, ${Object.keys(arrivals).length} ankomster`;


  const prefixes = Array.isArray(api.allowedMaterialPrefixes) ? api.allowedMaterialPrefixes.join(", ") : "";

  apiMeta.textContent = `updatedAt: ${api.updatedAt || "ukjent"} | tillatte materiellprefiks: ${prefixes || "ukjent"} | format: ${api.materialFormat || "ukjent"}`;


  const depRows = Object.entries(departures)

    .sort((a,b)=>{

      const ta = a[1] || "99:99";

      const tb = b[1] || "99:99";

      if(ta !== tb) return ta.localeCompare(tb, "no");

      return a[0].localeCompare(b[0], "no");

    });


  departuresTable.innerHTML = "<thead><tr><th>Tognr.</th><th>Avgang</th></tr></thead>";

  const depBody = document.createElement("tbody");

  depRows.forEach(([train, time])=>{

    const tr = document.createElement("tr");

    tr.innerHTML = `<td>${escapeHtml(train)}</td><td>${escapeHtml(time)}</td>`;

    depBody.appendChild(tr);

  });

  if(depRows.length === 0){

    const tr = document.createElement("tr");

    tr.innerHTML = "<td colspan='2'>Ingen avganger i API-data</td>";

    depBody.appendChild(tr);

  }

  departuresTable.appendChild(depBody);


  const arrRows = Object.entries(arrivals)

    .sort((a,b)=>{

      const ta = a[1]?.time || "99:99";

      const tb = b[1]?.time || "99:99";

      if(ta !== tb) return ta.localeCompare(tb, "no");

      return a[0].localeCompare(b[0], "no");

    });


  arrivalsTable.innerHTML = "<thead><tr><th>Tognr.</th><th>Ankomst</th><th>Neste dag</th></tr></thead>";

  const arrBody = document.createElement("tbody");

  arrRows.forEach(([train, info])=>{

    const tr = document.createElement("tr");

    tr.innerHTML = `<td>${escapeHtml(train)}</td><td>${escapeHtml(info.time || "")}</td><td>${info.nextDay ? "ja" : ""}</td>`;

    arrBody.appendChild(tr);

  });

  if(arrRows.length === 0){

    const tr = document.createElement("tr");

    tr.innerHTML = "<td colspan='3'>Ingen ankomster i API-data</td>";

    arrBody.appendChild(tr);

  }

  arrivalsTable.appendChild(arrBody);

}


async function fetchBaliseApi(){

  const apiStatus = document.getElementById("apiStatus");

  const apiMeta = document.getElementById("apiMeta");


  apiStatus.textContent = "Henter data fra /api/balise-vehicles ...";

  apiMeta.textContent = "Laster ...";


  try{

    const mode = state.baliseSelectedMode === "imorgen" ? "imorgen" : "idag";

    const trains = getRelevantTrainListForApi();

    const params = new URLSearchParams();

    params.set("mode", mode);

    if(trains.length){

      params.set("trains", trains.join(","));

    }


    const response = await fetch(`/api/balise-vehicles?${params.toString()}`, {cache:"no-store"});

    if(!response.ok){

      throw new Error(`HTTP ${response.status}`);

    }


    const payload = await response.json();

    const normalizedDepartures = normalizeApiDepartureMap(payload.departures);

    const normalizedArrivals = normalizeApiArrivalMap(payload.arrivals);

    const normalizedVehicles = buildVehicleMapFromPayload({vehicles: payload.vehicles || {}});


    state.apiBalise = {

      ok:Boolean(payload.ok),

      updatedAt:String(payload.updatedAt || ""),

      departures:normalizedDepartures,

      arrivals:normalizedArrivals,

      allowedMaterialPrefixes:Array.isArray(payload.allowedMaterialPrefixes) ? payload.allowedMaterialPrefixes.map(v=>String(v)) : [],

      materialFormat:Array.isArray(payload.materialFormat) ? payload.materialFormat.join(", ") : String(payload.materialFormat || ""),

      error:""

    };


    const meta = {

      source: payload.source || "balise.no",

      mode: payload.mode || mode,

      date: String(payload.date || ""),

      generated_at: String(payload.updatedAt || ""),

      hasDepartures: Object.keys(normalizedDepartures).length > 0

    };


    if((payload.mode || mode) === "imorgen"){

      state.baliseTomorrow = normalizedVehicles;

      state.baliseTomorrowDepartures = normalizedDepartures;

      state.baliseMetaTomorrow = meta;

    }else{

      state.baliseToday = normalizedVehicles;

      state.baliseTodayDepartures = normalizedDepartures;

      state.baliseMetaToday = meta;

    }


    persist(true);

    renderApiTables();

    updateBaliseStatus();

    updateBaliseViewState();

    buildOppstilling();

  }catch(err){

    state.apiBalise = {

      ok:false,

      updatedAt:"",

      departures:{},

      arrivals:{},

      allowedMaterialPrefixes:[],

      materialFormat:"",

      error:`API-feil: ${err.message}`

    };

    persist(true);

    renderApiTables();

    updateBaliseStatus();

    updateBaliseViewState();

    buildOppstilling();

  }

}
